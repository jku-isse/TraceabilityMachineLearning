/**
 * Hides the selected tasks.
 */
public void hideSelectedNodes() {
    DefaultMutableTreeNode[] t = getSelectedNodes();
    for (int i = 0; i < (t.length); i++) {
        DefaultMutableTreeNode parent = ((DefaultMutableTreeNode) (t[i].getParent()));
        hiddenTask.add(new GanttTree2.HiddenTask(t[i], parent, parent.getIndex(t[i])));
    }
    for (int i = 0; i < (hiddenTask.size()); i++) {
        GanttTree2.HiddenTask ht = ((GanttTree2.HiddenTask) (hiddenTask.get(i)));
        TreeNode parent = ht.node.getParent();
        if (parent != null)
            ((GanttTreeTableModel) (getTreeTable().getTreeTableModel())).removeNodeFromParent(((GanttTree2.HiddenTask) (hiddenTask.get(i))).getNode());

    }
}
