/* (non-Javadoc)

@see javax.swing.filechooser.FileFilter#accept(java.io.File)
 */
public boolean accept(File f) {
    if (f.isDirectory()) {
        return true;
    }
    String extension = GanttTXTFileFilter.getExtension(f);
    if ((extension != null) && (extension.equals("txt")))
        return true;

    return false;
}
