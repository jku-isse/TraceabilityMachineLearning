public int compareTo(Object o) {
    if (o == null)
        return 0;

    if (o instanceof GanttTreeTable.DisplayedColumn) {
        GanttTreeTable.DisplayedColumn dc = ((GanttTreeTable.DisplayedColumn) (o));
        if ((this.order) != (dc.order))
            return (this.order) - (dc.order);

        return this.id.compareTo(dc.id);
    }

}
