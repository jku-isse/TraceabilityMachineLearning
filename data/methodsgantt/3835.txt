static String getValueAsString(Object value) {
    final String result;
    if (value != null) {
        if (value instanceof GanttCalendar) {
            result = ((GanttCalendar) (value)).toXMLString();
        } else {
            result = String.valueOf(value);
        }
    } else {
        result = null;
    }
    return result;
}
