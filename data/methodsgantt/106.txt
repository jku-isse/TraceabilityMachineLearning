public void run() {
    StringBuffer buf = new StringBuffer();
    for (int i = 0; i < (myErrors.size()); i++) {
        buf.append(String.valueOf(myErrors.get(i)));
        buf.append("

");
    }
    myUIFacade.showErrorDialog(buf.toString());
    myErrors.clear();
    myUIFacade.resetErrorLog();
}
