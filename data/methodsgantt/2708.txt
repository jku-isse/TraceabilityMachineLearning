/**
 * Show the dialog box
 */
public File show() {
    File result = null;
    JFileChooser fc = new JFileChooser(myStartDirectory);
    fc.addChoosableFileFilter(new GanttXMLFileFilter());
    int returnVal = fc.showOpenDialog(null);
    if (returnVal == (JFileChooser.APPROVE_OPTION)) {
        result = fc.getSelectedFile();
    }
    return result;
}
