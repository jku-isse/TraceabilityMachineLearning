public void setValueAt(Object value, int row, int col) {
    if (row >= 0) {
        if (row >= (myAssignments.size())) {
            createAssignment(value);
        } else {
            updateAssignment(value, row, col);
        }
    } else {
        throw new IllegalArgumentException(("I cant set data in row=" + row));
    }
    isChanged = true;

}
