private String getFilePath(String value) {
    String result = null;
    String filePath;
    try {
        URL fileUrl = new URL(value);
        filePath = fileUrl.getPath();
    } catch (MalformedURLException e) {
        filePath = value;
    }
    if (new File(filePath).exists()) {
        result = filePath;
    }
    return result;
}
