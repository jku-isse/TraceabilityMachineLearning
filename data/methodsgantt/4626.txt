public int compareTo(Object o) {
    SearchKey rvalue = ((SearchKey) (o));
    int result = (myFirstTaskID) - (rvalue.myFirstTaskID);
    if (result == 0) {
        result = (myType) - (rvalue.myType);
    }
    if (result == 0) {
        result = (mySecondTaskID) - (rvalue.mySecondTaskID);
    }
    return result;
}
