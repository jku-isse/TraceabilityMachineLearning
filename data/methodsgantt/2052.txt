private void createRightSideText(TaskActivity taskActivity) {
    Rectangle bounds = getBoundingRectangle(taskActivity);
    String text = "";
    int xText;
    int yText;
    text = getTaskLabel(taskActivity.getTask(), TaskRendererImpl.RIGHT);
    xText = ((int) (bounds.getMaxX())) + 9;
    yText = ((int) (myModel.getBoundingRectangle(taskActivity.getTask()).getMaxY())) - 3;
    GraphicPrimitiveContainer.Text textPrimitive = processText(xText, yText, text);
}
