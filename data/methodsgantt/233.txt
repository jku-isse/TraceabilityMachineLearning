public void repaint() {
    try {
        if (((myChartModel) != null) && ((myTableHeader) != null)) {
            myChartModel.setHeaderHeight(((myTableHeader.getHeight()) + (GanttGraphicArea.HEADER_OFFSET)));
        }
    } catch (NullPointerException e) {
        if (!(GPLogger.log(e))) {
            e.printStackTrace(System.err);
        }
    }
    super.repaint();
}
