/* JTable uses this method to determine the default renderer/ editor for
each cell. If we didnt implement this method, then the last column would
contain text ("true"/"false"), rather than a check box.
 */
public Class getColumnClass(int c) {
    // if (c == 0 || c == 2) {
    // return String.class;
    // } else {
    // return HumanResource.class;
    // }
    switch (c) {
        case 0 :
        case 2 :
            return String.class;
        case 1 :
            return HumanResource.class;
        case 3 :
            return Boolean.class;
        case 4 :
            return Role.class;
        default :
            return String.class;
    }
}
