public void setVisible(boolean isVisible) {
    String tabName = "New view";
    JScrollPane jsp = new JScrollPane();
    if (isVisible) {
        myChart.setTaskManager(this.getTaskManager());
        tabName = myChart.getName();
        jsp.getViewport().add(myComponent);
        myTabs.addTab(tabName, myIcon, jsp, tabName, this);
        myTabs.setSelectedComponent(jsp);
        myIndex = myTabs.getSelectedIndex();
    } else {
        myTabs.remove(myIndex);
    }
    this.isVisible = isVisible;
}
