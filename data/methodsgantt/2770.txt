public boolean hasChanges() {
    Role[] roles = getRoleManager().getProjectLevelRoles();
    for (int i = 0; i < (getRowCount()); i++) {
        String nextRoleName = ((String) (getValueAt(i, 1)));
        if (nextRoleName != null) {
            if (i >= (roles.length))
                return true;

            if (!(nextRoleName.equals(roles[i].getName())))
                return true;

        }
    }
    return false;
}
