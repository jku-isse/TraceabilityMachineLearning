public NewTaskAction(IGanttProject project) {
    myProject = project;
    setText(project.getI18n());
    putValue(Action.ACCELERATOR_KEY, KeyStroke.getKeyStroke(KeyEvent.VK_T, Toolkit.getDefaultToolkit().getMenuShortcutKeyMask()));
    URL iconUrl = this.getClass().getClassLoader().getResource("icons/insert_16.gif");
    if (iconUrl != null) {
        this.putValue(Action.SMALL_ICON, new ImageIcon(iconUrl));
    }
    project.getI18n().addListener(this);
}
