public ProjectResource newHumanResource() {
    final HumanResource people = ((HumanResourceManager) (getHumanResourceManager())).newHumanResource();
    people.setRole(getRoleManager().getDefaultRole());
    GanttDialogPerson dp = new GanttDialogPerson(getUIFacade(), getLanguage(), people);
    dp.setVisible(true);
    if (dp.result()) {
        getUndoManager().undoableEdit("new Resource", new Runnable() {
            public void run() {
                getHumanResourceManager().add(people);
            }
        });
    }
    return people;
}
