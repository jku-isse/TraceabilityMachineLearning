/**
 * Write the header of the XFIG FILE Based on the xfig file specification.
 */
public void xfigheader(OutputStreamWriter fout) throws IOException {
    if (debug)
        System.out.println("xfigheader begin");

    fout.write("#FIG 3.2
");// version

    fout.write("Landscape
");// orientation

    fout.write("Center
");// justification

    fout.write("Inches
");// units

    fout.write("Letter
");// papersize

    fout.write("100.0
");// magnification

    fout.write("Single
");// multiplePage

    fout.write("-2
");// transparentColor

    fout.write("1200 2
");// resolutionPpi origin

    if (debug)
        System.out.println("xfigheader end");

}
