public GraphicPrimitiveContainer.GraphicPrimitive getPrimitive(int x, int xThreshold, int y, int yThreshold) {
    for (int i = 0; i < (myRectangles.size()); i++) {
        GraphicPrimitiveContainer.Rectangle next = ((GraphicPrimitiveContainer.Rectangle) (myRectangles.get(i)));
        // System.err.println(" next rectangle="+next);
        if (((((next.myLeftX) <= (x + xThreshold)) && (((next.myLeftX) + (next.myWidth)) >= (x - xThreshold))) && ((next.myTopY) <= (y + yThreshold))) && (((next.myTopY) + (next.myHeight)) >= (y - yThreshold))) {
            return next;
        }
    }
    return null;
}
