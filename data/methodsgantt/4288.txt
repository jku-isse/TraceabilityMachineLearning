public TaskMutator createMutatorFixingDuration() {
    if ((myMutator) != null) {
        throw new RuntimeException(("Two mutators have been requested for task=" + (getName())), myException);
    }
    myMutator = new TaskImpl.MutatorImpl() {
        public void setStart(GanttCalendar start) {
            super.setStart(start);
            TaskImpl.this.myEnd = null;
        }
    };
    myException = new Exception();
    return myMutator;
}
