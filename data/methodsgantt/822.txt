/**
 * Returne the mother task.
 */
public DefaultMutableTreeNode getFatherNode(Task node) {
    if (node == null) {
        return null;
    }
    DefaultMutableTreeNode tmp = ((DefaultMutableTreeNode) (getNode(node.getTaskID())));
    if (tmp == null) {
        return null;
    }
    return ((DefaultMutableTreeNode) (tmp.getParent()));
}
