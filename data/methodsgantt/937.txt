private void hideColumn(String name) {
    Iterator it = mapTableColumnColumnKeeper.keySet().iterator();
    while (it.hasNext()) {
        TableColumn c = ((TableColumn) (it.next()));
        String n = ((String) (c.getHeaderValue()));
        if (n.equals(name)) {
            GanttTreeTable.ColumnKeeper ck = ((GanttTreeTable.ColumnKeeper) (mapTableColumnColumnKeeper.get(c)));
            if (ck.isShown)
                ck.hide();

            break;
        }
    } 
    // createPopupMenu();
}
