public ResourceManager getHumanResourceManager() {
    ResourceManager result = ((ResourceManager) (managerHash.get(GanttProject.HUMAN_RESOURCE_MANAGER_ID)));
    if (result == null) {
        result = new HumanResourceManager(getRoleManager().getDefaultRole());
        // result.addView(getPeople());
        managerHash.put(GanttProject.HUMAN_RESOURCE_MANAGER_ID, result);
        result.addView(this);
    }
    return result;
}
