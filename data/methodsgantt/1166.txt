/**
 * Changes the language.
 *
 * @param ganttLanguage
 * 		New language to use.
 */
public void changeLanguage(GanttLanguage ganttLanguage) {
    ResourceTreeTableModel.strResourceName = ResourceTreeTableModel.language.getText("tableColResourceName");
    ResourceTreeTableModel.strResourceRole = ResourceTreeTableModel.language.getText("tableColResourceRole");
    ResourceTreeTableModel.strResourceEMail = ResourceTreeTableModel.language.getText("tableColResourceEMail");
    ResourceTreeTableModel.strResourcePhone = ResourceTreeTableModel.language.getText("tableColResourcePhone");
    ResourceTreeTableModel.strResourceRoleForTask = ResourceTreeTableModel.language.getText("tableColResourceRoleForTask");
    // hack assume that INDEX_RESOURCE_ROLE_TASK is the last index
    String[] cols = new String[(ResourceTreeTableModel.INDEX_RESOURCE_ROLE_TASK) + 1];
    cols[ResourceTreeTableModel.INDEX_RESOURCE_EMAIL] = ResourceTreeTableModel.strResourceEMail;
    cols[ResourceTreeTableModel.INDEX_RESOURCE_NAME] = ResourceTreeTableModel.strResourceName;
    cols[ResourceTreeTableModel.INDEX_RESOURCE_PHONE] = ResourceTreeTableModel.strResourcePhone;
    cols[ResourceTreeTableModel.INDEX_RESOURCE_ROLE] = ResourceTreeTableModel.strResourceRole;
    cols[ResourceTreeTableModel.INDEX_RESOURCE_ROLE_TASK] = ResourceTreeTableModel.strResourceRoleForTask;
    for (int i = 0; i < (cols.length); i++) {
        ResourceColumn col = ((ResourceColumn) (columns.get(new Integer(i))));
        if (col != null)
            col.setTitle(cols[i]);

    }
}
