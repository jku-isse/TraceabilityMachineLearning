public void deleteDependency(TaskDependency dependency) {
    TaskDependencyCollectionImpl.MutationInfo info = ((TaskDependencyCollectionImpl.MutationInfo) (myQueue.get(dependency)));
    if (info == null) {
        myQueue.put(dependency, new TaskDependencyCollectionImpl.MutationInfo(dependency, TaskDependencyCollectionImpl.MutationInfo.DELETE));
    } else
        if ((info.myOperation) == (TaskDependencyCollectionImpl.MutationInfo.ADD)) {
            myQueue.remove(dependency);
        }

}
