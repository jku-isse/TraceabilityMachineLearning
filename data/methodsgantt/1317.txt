public static KeyStroke getKeyStroke(String keystrokeID) {
    if ((GPAction.ourProperties) == null) {
        GPAction.ourProperties = new Properties();
        URL url = GPAction.class.getResource("/keyboard.properties");
        if (url == null) {
            return null;
        }
        URL resolvedUrl;
        try {
            resolvedUrl = Platform.resolve(url);
            GPAction.ourProperties.load(resolvedUrl.openStream());
        } catch (IOException e) {
            if (!(GPLogger.log(e))) {
                e.printStackTrace(System.err);
            }
            return null;
        }
    }
    String stringKeyStroke = ((String) (GPAction.ourProperties.get(keystrokeID)));
    return KeyStroke.getKeyStroke(stringKeyStroke);
}
