/**
 * Send an Email to the current resource
 */
public void sendMail(GanttProject parent) {
    if ((((table) != null) && ((table.getSelectedNodes()) != null)) && ((table.getSelectedNodes().length) > 0)) {
        HumanResource people = ((HumanResource) (table.getSelectedNodes()[0].getUserObject()));
        if (people != null) {
            try {
                BrowserControl.displayURL(("mailto:" + (people.getMail())));
            } catch (Exception e) {
                System.err.println(e);
            }
        }
    } else {
        GanttDialogInfo gdi = new GanttDialogInfo(appli, GanttDialogInfo.INFO, GanttDialogInfo.YES_OPTION, GanttLanguage.getInstance().getText("msg26"), GanttLanguage.getInstance().getText("sendMail"));
        gdi.show();
    }
}
