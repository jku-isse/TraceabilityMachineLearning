public void commit() {
    List mutations = new ArrayList(myQueue.values());
    if ((myCleanupMutation) != null) {
        mutations.add(myCleanupMutation);
    }
    Collections.sort(mutations);
    for (int i = 0; i < (mutations.size()); i++) {
        TaskDependencyCollectionImpl.MutationInfo next = ((TaskDependencyCollectionImpl.MutationInfo) (mutations.get(i)));
        switch (next.myOperation) {
            case TaskDependencyCollectionImpl.MutationInfo.ADD :
                {
                    try {
                        addDependency(next.myDependency);
                    } catch (TaskDependencyException e) {
                        if (!(GPLogger.log(e))) {
                            e.printStackTrace(System.err);
                        }
                    }
                    break;
                }
            case TaskDependencyCollectionImpl.MutationInfo.DELETE :
                {
                    delete(next.myDependency);
                    break;
                }
            case TaskDependencyCollectionImpl.MutationInfo.CLEAR :
                {
                    doClear();
                    break;
                }
        }
    }
}
