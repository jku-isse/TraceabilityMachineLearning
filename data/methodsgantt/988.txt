public void add(String name, int order, int width) {
    GanttTreeTable.DisplayedColumn newColumn = new GanttTreeTable.DisplayedColumn(name);
    newColumn.setOrder(order);
    if (width >= 0) {
        newColumn.setWidth(width);
    }
    newColumn.setDisplayed(true);
    GanttTreeTable.DisplayedColumnsList clone = ((GanttTreeTable.DisplayedColumnsList) (getDisplayColumns().clone()));
    clone.add(newColumn);
    setDisplayedColumns(clone);
}
