public void registerTask(Task task) {
    int taskID = task.getTaskID();
    if ((myTaskMap.getTask(taskID)) == null) {
        // if the taskID is
        // not in the map
        myTaskMap.addTask(task);
        if ((getMaxID()) < taskID) {
            setMaxID((taskID + 1));
        }
    } else {
        // taskID has been in the map. the newTask will not be added
        throw new RuntimeException(("There is a task that already has the ID " + taskID));
    }
}
