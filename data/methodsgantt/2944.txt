public Object getValueAt(int row, int col) {
    AboutLibraryPanel.LibraryInfo info = ((AboutLibraryPanel.LibraryInfo) (data.elementAt(row)));
    if (col == 0)
        return info.getName();
    else
        if (col == 1)
            return info.getWeb();
        else
            if (col == 2)
                return info.getComment();
            else
                return null;



}
