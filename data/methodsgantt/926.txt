/**
 * Displays all the table columns.
 */
private void displayAllColumns() {
    Iterator it = mapTableColumnColumnKeeper.values().iterator();
    while (it.hasNext()) {
        GanttTreeTable.ColumnKeeper ck = ((GanttTreeTable.ColumnKeeper) (it.next()));
        if (!(ck.isShown))
            ck.show();

    } 
    // createPopupMenu();
}
