/**
 * Creates a polygon array of the bezier path.
 *
 * @return Point array.
 */
public Point2D.Double[] toPolygonArray() {
    Point2D.Double[] points = new Point2D.Double[size()];
    for (int i = 0, n = size(); i < n; i++) {
        points[i] = new Point2D.Double(get(i).x[0], get(i).y[0]);
    }
    return points;
}
