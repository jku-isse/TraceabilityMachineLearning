/**
 * Displays an option pane as a sheet on its parent window.
 *
 * @param pane
 * 		The option pane.
 * @param parentComponent
 * 		The parent of the option pane.
 * @param listener
 * 		The listener for SheetEvents.
 */
public static void showSheet(JOptionPane pane, Component parentComponent, SheetListener listener) {
    final JSheet sheet = JSheet.createSheet(pane, parentComponent, JSheet.styleFromMessageType(pane.getMessageType()));
    sheet.addSheetListener(listener);
    sheet.show();
}
