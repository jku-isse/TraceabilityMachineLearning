protected void fileOpened(final View view, File file, Object value) {
    if (value == null) {
        view.setFile(file);
        view.setEnabled(true);
        getApplication().addRecentFile(file);
    } else {
        JSheet.showMessageSheet(view.getComponent(), ((((("<html>" + (UIManager.getString("OptionPane.css"))) + "<b>Couldn	 open the file "") + file) + "".</b><br>") + value), JOptionPane.ERROR_MESSAGE, new SheetListener() {
            public void optionSelected(SheetEvent evt) {
                view.clear();
                view.setEnabled(true);
            }
        });
    }
}
