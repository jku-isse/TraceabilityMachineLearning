/**
 * Constructs a <code>Dimension2DDouble</code> and initializes
 * it to the specified width and specified height.
 *
 * @param width
 * 		the specified width
 * @param height
 * 		the specified height
 */
public Dimension2DDouble(double width, double height) {
    this.width = width;
    this.height = height;
}
