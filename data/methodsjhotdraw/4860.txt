protected GeneralPath getPath() {
    if ((cachedPath) == null) {
        cachedPath = new GeneralPath();
        cachedPath.setWindingRule(((AttributeKeys.WINDING_RULE.get(this)) == (AttributeKeys.WindingRule.EVEN_ODD) ? GeneralPath.WIND_EVEN_ODD : GeneralPath.WIND_NON_ZERO));
        for (Figure child : getChildren()) {
            ODGBezierFigure b = ((ODGBezierFigure) (child));
            cachedPath.append(b.getBezierPath(), false);
        }
    }
    return cachedPath;
}
