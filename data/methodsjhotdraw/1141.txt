/**
 * Convenience method for computing the total stroke miter limit from the
 * STROKE_MITER_LIMIT, and IS_STROKE_MITER_LIMIT factor.
 */
public static double getStrokeTotalMiterLimit(Figure f) {
    if (AttributeKeys.IS_STROKE_MITER_LIMIT_FACTOR.get(f)) {
        return (AttributeKeys.STROKE_MITER_LIMIT.get(f)) * (AttributeKeys.STROKE_WIDTH.get(f));
    } else {
        return AttributeKeys.STROKE_MITER_LIMIT.get(f);
    }
}
