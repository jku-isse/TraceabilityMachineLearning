protected Point2D.Double getOrigin() {
    // This handle is placed above the figure.
    // We move it up by a handlesizes, so that it wont overlap with
    // the handles from TransformHandleKit.
    Rectangle2D.Double bounds = getTransformedBounds();
    Point2D.Double origin = new Point2D.Double(bounds.getCenterX(), ((bounds.y) - ((getHandlesize()) / (view.getScaleFactor()))));
    return origin;
}
