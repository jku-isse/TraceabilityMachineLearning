/**
 * Convenience method for setting a value on the
 * specified figure and calling willChange before and changed
 * after setting the value.
 *
 * @param f
 * 		the Figure
 * @param value
 * 		the attribute value
 */
public void set(Figure f, T value) {
    f.willChange();
    basicSet(f, value);
    f.changed();
}
