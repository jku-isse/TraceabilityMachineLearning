public double getDecorationRadius(Figure f) {
    double strokeWidth = AttributeKeys.STROKE_WIDTH.get(f);
    double scaleFactor;
    if (strokeWidth > 1.0F) {
        scaleFactor = 1.0 + ((strokeWidth - 1.0) / 2.0);
    } else {
        scaleFactor = 1.0;
    }
    return (getDecoratorPathRadius(f)) * scaleFactor;
}
