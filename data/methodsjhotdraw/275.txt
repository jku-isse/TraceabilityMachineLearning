/**
 * Creates a new instance.
 */
public ArrangeAction(Arrangeable arrangeable, Arrangeable.Arrangement arrangement) {
    this.arrangeable = arrangeable;
    this.arrangement = arrangement;
    ResourceBundleUtil labels = ResourceBundleUtil.getBundle("org.jhotdraw.app.Labels");
    String labelID;
    switch (arrangement) {
        case VERTICAL :
            labelID = ArrangeAction.VERTICAL_ID;
            break;
        case HORIZONTAL :
            labelID = ArrangeAction.HORIZONTAL_ID;
            break;
        case CASCADE :
        default :
            labelID = ArrangeAction.CASCADE_ID;
            break;
    }
    labels.configureAction(this, labelID);
}
