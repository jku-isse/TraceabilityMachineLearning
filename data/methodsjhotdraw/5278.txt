public CombineAction(DrawingEditor editor, SVGPathFigure prototype, boolean isGroupingAction) {
    super(editor);
    this.prototype = prototype;
    this.isCombineAction = isGroupingAction;
    labels = ResourceBundleUtil.getBundle("org.jhotdraw.samples.svg.Labels", Locale.getDefault());
    labels.configureAction(this, CombineAction.ID);
}
