private void writeHrefAttribute(IXMLElement elem, SVGFigure f) {
    if (((SVGAttributeKeys.LINK.get(f)) != null) && ((SVGAttributeKeys.LINK.get(f).trim().length()) > 0)) {
        elem.setAttribute("href", SVGAttributeKeys.LINK.get(f));
        elem.setAttribute("title", SVGAttributeKeys.LINK.get(f));
        elem.setAttribute("alt", SVGAttributeKeys.LINK.get(f));
        if (((SVGAttributeKeys.LINK_TARGET.get(f)) != null) && ((SVGAttributeKeys.LINK_TARGET.get(f).trim().length()) > 0)) {
            elem.setAttribute("target", SVGAttributeKeys.LINK_TARGET.get(f));
        }
    } else {
        elem.setAttribute("nohref", "true");
    }
}
