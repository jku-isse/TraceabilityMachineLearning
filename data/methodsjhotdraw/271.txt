/**
 * Updates the listeners of this action depending on the current view
 * of the application.
 */
protected void updateView(View oldValue, View newValue) {
    if (oldValue != null) {
        uninstallViewListeners(oldValue);
    }
    if (newValue != null) {
        installViewListeners(newValue);
    }
    firePropertyChange(AbstractViewAction.VIEW_PROPERTY, oldValue, newValue);
    updateEnabled(((oldValue != null) && (oldValue.isEnabled())), ((newValue != null) && (newValue.isEnabled())));
    updateView();
}
