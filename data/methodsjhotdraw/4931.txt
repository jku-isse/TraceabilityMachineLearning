/**
 * Adds a cubic curve to the current path segment.
 * This is only allowed, when the current path segment is open.
 */
public void curveTo(Object x1, Object y1, Object x2, Object y2, Object x3, Object y3) {
    if (((size()) == 0) || ((get(((size()) - 1)).type) == (EnhancedPath.SegType.CLOSE))) {
        throw new IllegalPathStateException("curveTo is only allowed when a path segment is open");
    }
    add(new EnhancedPath.Segment(EnhancedPath.SegType.CURVETO, x1, y1, x2, y2, x3, y3));
}
