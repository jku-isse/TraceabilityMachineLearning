/**
 * Scales the input vector to the new length and returns it.
 * <p>
 * This method alters the value of the input point!
 */
private static Point2D.Double v2Scale(Point2D.Double v, double newlen) {
    double len = Bezier.v2Length(v);
    if (len != 0.0) {
        v.x *= newlen / len;
        v.y *= newlen / len;
    }
    return v;
}
