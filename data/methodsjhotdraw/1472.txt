@Override
public void draw(Graphics2D g) {
    Graphics2D gg = ((Graphics2D) (g.create()));
    gg.transform(getView().getDrawingToViewTransform());
    if ((targetFigure) != null) {
        for (Connector c : targetFigure.getConnectors(getPrototype())) {
            c.draw(gg);
        }
    }
    if ((createdFigure) != null) {
        createdFigure.draw(gg);
        Point p = getView().drawingToView(createdFigure.getStartPoint());
        Ellipse2D.Double e = new Ellipse2D.Double(((p.x) - ((ConnectionTool.ANCHOR_WIDTH) / 2)), ((p.y) - ((ConnectionTool.ANCHOR_WIDTH) / 2)), ConnectionTool.ANCHOR_WIDTH, ConnectionTool.ANCHOR_WIDTH);
        g.setColor(Color.GREEN);
        g.fill(e);
        g.setColor(Color.BLACK);
        g.draw(e);
        p = getView().drawingToView(createdFigure.getEndPoint());
        e = new Ellipse2D.Double(((p.x) - ((ConnectionTool.ANCHOR_WIDTH) / 2)), ((p.y) - ((ConnectionTool.ANCHOR_WIDTH) / 2)), ConnectionTool.ANCHOR_WIDTH, ConnectionTool.ANCHOR_WIDTH);
        g.setColor(Color.GREEN);
        g.fill(e);
        g.setColor(Color.BLACK);
        g.draw(e);
    }
    gg.dispose();
}
