/**
 * Sets the reader for the tokenizer.
 *
 * @param r
 * 		
 */
public void setReader(Reader r) {
    this.reader = r;
    readpos = 0;
    unread.clear();
    peekc = StreamPosTokenizer.NEED_CHAR;
    pushedBack = false;
    forceLower = false;
    lineno = 0;
    startpos = endpos = -1;
    ttype = StreamPosTokenizer.TT_NOTHING;
}
