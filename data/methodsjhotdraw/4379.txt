/**
 * Flushes and closes (I think, in the superclass) the stream.
 *
 * @since 1.3
 */
public void close() throws IOException {
    // 1. Ensure that pending characters are written
    flushBase64();
    // 2. Actually close the stream
    // Base class both flushes and closes.
    super.close();
    buffer = null;
    out = null;
}// end close
