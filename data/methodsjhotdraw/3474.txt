/**
 * Returns the direction OUT_TOP, OUT_BOTTOM, OUT_LEFT, OUT_RIGHT from
 * one point to another one.
 */
public static int direction(double x1, double y1, double x2, double y2) {
    int direction = 0;
    double vx = x2 - x1;
    double vy = y2 - y1;
    if ((vy < vx) && (vx > (-vy))) {
        direction = Geom.OUT_RIGHT;
    } else
        if ((vy > vx) && (vy > (-vx))) {
            direction = Geom.OUT_TOP;
        } else
            if ((vx < vy) && (vx < (-vy))) {
                direction = Geom.OUT_LEFT;
            } else {
                direction = Geom.OUT_BOTTOM;
            }


    return direction;
}
