public Figure findFigureBehind(Point2D.Double p, Collection<? extends Figure> children) {
    int inFrontOf = children.size();
    for (Figure f : getFiguresFrontToBack()) {
        if (inFrontOf == 0) {
            if ((f.isVisible()) && (f.contains(p))) {
                return f;
            }
        } else {
            if (children.contains(f)) {
                inFrontOf--;
            }
        }
    }
    return null;
}
