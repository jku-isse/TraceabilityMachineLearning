/**
 * Sets the active view. Calls deactivate on the previously
 * active view, and then calls activate on the given view.
 *
 * @param newValue
 * 		Active view, can be null.
 */
public void setActiveView(View newValue) {
    View oldValue = activeView;
    if ((activeView) != null) {
        activeView.deactivate();
    }
    activeView = newValue;
    if ((activeView) != null) {
        activeView.activate();
    }
    firePropertyChange(Application.ACTIVE_VIEW_PROPERTY, oldValue, newValue);
}
