/**
 * Unreads a character back into the input stream of the tokenizer.
 */
public void pushCharBack(int ch) throws IOException {
    if (pushedBack) {
        throw new IllegalStateException("cant push back char when a token has been pushed back");
    }
    if ((peekc) == (StreamPosTokenizer.NEED_CHAR)) {
        unread(ch);
    } else {
        unread(peekc);
        peekc = StreamPosTokenizer.NEED_CHAR;
        unread(ch);
    }
}
