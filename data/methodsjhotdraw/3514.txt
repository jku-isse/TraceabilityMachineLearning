/**
 * Adds the current insets to the specified Rectangle2D making the
 * rectangle larger.
 *
 * @param r
 * 		The Rectangle2D.
 */
public void addTo(Rectangle2D r) {
    r.setRect(((r.getX()) - (getLeft())), ((r.getY()) - (getTop())), (((r.getWidth()) + (getLeft())) + (getRight())), (((r.getHeight()) + (getTop())) + (getBottom())));
}
