@Override
protected void updateView(View oldValue, View newValue) {
    super.updateView(oldValue, newValue);
    if ((newValue != null) && ((newValue.getAction(RedoAction.ID)) != null)) {
        putValue(AbstractAction.NAME, newValue.getAction(RedoAction.ID).getValue(AbstractAction.NAME));
        updateEnabledState();
    }
}
