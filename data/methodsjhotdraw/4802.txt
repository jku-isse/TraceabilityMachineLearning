public Rectangle2D.Double getDrawingArea() {
    if ((cachedDrawingArea) == null) {
        if ((AttributeKeys.TRANSFORM.get(this)) == null) {
            cachedDrawingArea = path.getBounds2D();
        } else {
            BezierPath p2 = ((BezierPath) (path.clone()));
