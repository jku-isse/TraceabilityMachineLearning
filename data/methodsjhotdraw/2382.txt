/**
 * Creates handles for each lead of a
 * figure and adds them to the provided collection.
 */
public static void addLeadHandles(Figure f, Collection<Handle> handles) {
    handles.add(new NullHandle(f, new RelativeLocator(0.0F, 0.0F)));
    handles.add(new NullHandle(f, new RelativeLocator(0.0F, 1.0F)));
    handles.add(new NullHandle(f, new RelativeLocator(1.0F, 0.0F)));
    handles.add(new NullHandle(f, new RelativeLocator(1.0F, 1.0F)));
}
