 * rollover border will be installed if rollover borders are
 * enabled.

 *
 * @since 1.4
 */
protected Border createRolloverBorder() {

    if (border != null) {
        return ((Border) (border));
    }
    UIDefaults table = UIManager.getLookAndFeelDefaults();
