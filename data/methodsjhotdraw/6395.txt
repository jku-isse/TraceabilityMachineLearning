/**
 * Adds an Enum value to the DOMFactory.
 */
public void addEnum(String value, Enum e) {
    DefaultDOMFactory.enumToValueMap.put(e, value);
    Set<Enum> enums;
    if (DefaultDOMFactory.valueToEnumMap.containsKey(value)) {
        enums = DefaultDOMFactory.valueToEnumMap.get(value);
    } else {
        enums = new HashSet<Enum>();
        DefaultDOMFactory.valueToEnumMap.put(value, enums);
    }
    enums.add(e);
}
