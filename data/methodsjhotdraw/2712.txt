private Rectangle2D.Double getFieldBounds(TextHolderFigure figure) {
    Rectangle2D.Double r = figure.getDrawingArea();
    Insets2D.Double insets = figure.getInsets();
    insets.subtractTo(r);
    // FIXME - Find a way to determine the parameters for grow.
    // r.grow(1,2);
    // r.width += 16;
    r.x -= 1;
    r.y -= 2;
    r.width += 18;
    r.height += 4;
    return r;
}
