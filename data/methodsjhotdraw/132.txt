/**
 * Creates a menu bar.
 */
protected JMenuBar createMenuBar() {
    JMenuBar mb = new JMenuBar();
    mb.add(createFileMenu());
    for (JMenu mm : getModel().createMenus(this, null)) {
        mb.add(mm);
    }
    mb.add(createWindowMenu());
    mb.add(createHelpMenu());
    return mb;
}
