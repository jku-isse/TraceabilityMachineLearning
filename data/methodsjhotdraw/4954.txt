public Object clone() {
    try {
        EnhancedPath.Segment that = ((EnhancedPath.Segment) (super.clone()));
        that.coords = this.coords.clone();
        that.modifiers = this.modifiers.clone();
        return that;
    } catch (CloneNotSupportedException e) {
        InternalError error = new InternalError();
        error.initCause(e);
        throw error;
    }
}
