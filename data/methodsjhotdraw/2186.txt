/**
 * Sets the image.
 *
 * @param imageData
 * 		The image data. If this is null, a buffered image must
 * 		be provided.
 * @param bufferedImage
 * 		An image constructed from the imageData. If this
 * 		is null, imageData must be provided.
 */
public void setImage(byte[] imageData, BufferedImage bufferedImage) {
    willChange();
    this.imageData = imageData;
    this.bufferedImage = bufferedImage;
    changed();
}
