/**
 * Translates the vertices of the <code>Polygon</code> by
 * <code>deltaX</code> along the x axis and by
 * <code>deltaY</code> along the y axis.
 *
 * @param deltaX
 * 		the amount to translate along the X axis
 * @param deltaY
 * 		the amount to translate along the Y axis
 */
@Override
public void translate(double deltaX, double deltaY) {
    for (int i = 0; i < (npoints); i++) {
        xpoints[i] += deltaX;
        ypoints[i] += deltaY;
    }
    if ((bounds) != null) {
        bounds.x += deltaX;
        bounds.y += deltaY;
    }
}
