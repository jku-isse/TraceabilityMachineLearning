public Figure findFigureBehind(Point2D.Double p, Collection<Figure> figures) {
    int inFrontOf = figures.size();
    for (Figure f : getFiguresFrontToBack()) {
        if (inFrontOf == 0) {
            if ((f.isVisible()) && (f.contains(p))) {
                return f;
            }
        } else {
            if (figures.contains(f)) {
                inFrontOf--;
            }
        }
    }
    return null;
}
