protected Icon getIcon(Component c) {
    Icon icon;
    boolean isActive = true;// QuaquaUtilities.isOnActiveWindow(c);

    if (c instanceof JSlider) {
        JSlider slider = ((JSlider) (c));
        if (isActive) {
            if (c.isEnabled()) {
                if (slider.getModel().getValueIsAdjusting()) {
                    icon = icons[SliderThumbIcon.EP];
                } else {
                    icon = icons[SliderThumbIcon.E];
                }
            } else {
                icon = icons[SliderThumbIcon.D];
            }
        } else {
            if (c.isEnabled()) {
                icon = icons[SliderThumbIcon.I];
            } else {
                icon = icons[SliderThumbIcon.DI];
            }
        }
    } else {
        if (isActive) {
            if (c.isEnabled()) {
                icon = icons[SliderThumbIcon.E];
            } else {
                icon = icons[SliderThumbIcon.D];
            }
        } else {
            if (c.isEnabled()) {
                icon = icons[SliderThumbIcon.I];
            } else {
                icon = icons[SliderThumbIcon.DI];
            }
        }
    }
    return icon;
}
