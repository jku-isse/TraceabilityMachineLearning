protected Point getLocation() {
    if ((getOwner().getNodeCount()) > (index)) {
        Point2D.Double p = getOwner().getPoint(index, 0);
        if ((AttributeKeys.TRANSFORM.get(getTransformOwner())) != null) {
            AttributeKeys.TRANSFORM.get(getTransformOwner()).transform(p, p);
        }
        return view.drawingToView(p);
    } else {
        return new Point(10, 10);
    }
}
