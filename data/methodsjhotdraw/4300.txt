public void actionPerformed(ActionEvent evt) {
    String key = getName();
    JToolBar toolBar = ((JToolBar) (evt.getSource()));
    PaletteToolBarUI ui = ((PaletteToolBarUI) (PaletteLookAndFeel.getUIOfType(toolBar.getUI(), PaletteToolBarUI.class)));
    if ((PaletteToolBarUI.Actions.NAVIGATE_RIGHT) == key) {
        ui.navigateFocusedComp(SwingConstants.EAST);
    } else
        if ((PaletteToolBarUI.Actions.NAVIGATE_LEFT) == key) {
            ui.navigateFocusedComp(SwingConstants.WEST);
        } else
            if ((PaletteToolBarUI.Actions.NAVIGATE_UP) == key) {
                ui.navigateFocusedComp(SwingConstants.NORTH);
            } else
                if ((PaletteToolBarUI.Actions.NAVIGATE_DOWN) == key) {
                    ui.navigateFocusedComp(SwingConstants.SOUTH);
                }



}
