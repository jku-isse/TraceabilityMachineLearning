/**
 * post the POST request to the server, with the specified parameter
 *
 * @param name
 * 		parameter name
 * @param value
 * 		parameter value
 * @return input stream with the server response
 * @throws IOException
 * 		
 * @see #setParameter
 */
public InputStream post(String name, Object value) throws IOException {
    postCookies();
    setParameter(name, value);
    return doPost();
}
