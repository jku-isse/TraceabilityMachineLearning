/**
 * This method returns an image object that can be used to
 * represent the bean in toolboxes, toolbars, etc.   Icon images
 * will typically be GIFs, but may in future include other formats.
 * <p>
 * Beans arent required to provide icons and may return null from
 * this method.
 * <p>
 * There are four possible flavors of icons (16x16 color,
 * 32x32 color, 16x16 mono, 32x32 mono).  If a bean choses to only
 * support a single icon we recommend supporting 16x16 color.
 * <p>
 * We recommend that icons have a "transparent" background
 * so they can be rendered onto an existing background.
 *
 * @param iconKind
 * 		The kind of icon requested.  This should be
 * 		one of the constant values ICON_COLOR_16x16, ICON_COLOR_32x32,
 * 		ICON_MONO_16x16, or ICON_MONO_32x32.
 * @return An image object representing the requested icon.  May
return null if no suitable icon is available.
 */
public Image getIcon(int iconKind) {
    switch (iconKind) {
        case BeanInfo.ICON_COLOR_16x16 :
            if ((JLifeFormattedTextFieldBeanInfo.iconNameC16) == null) {
                return null;
            } else {
                if ((JLifeFormattedTextFieldBeanInfo.iconColor16) == null) {
                    JLifeFormattedTextFieldBeanInfo.iconColor16 = loadImage(JLifeFormattedTextFieldBeanInfo.iconNameC16);
                }
                return JLifeFormattedTextFieldBeanInfo.iconColor16;
            }
        case BeanInfo.ICON_COLOR_32x32 :
            if ((JLifeFormattedTextFieldBeanInfo.iconNameC32) == null) {
                return null;
            } else {
                if ((JLifeFormattedTextFieldBeanInfo.iconColor32) == null) {
                    JLifeFormattedTextFieldBeanInfo.iconColor32 = loadImage(JLifeFormattedTextFieldBeanInfo.iconNameC32);
                }
                return JLifeFormattedTextFieldBeanInfo.iconColor32;
            }
        case BeanInfo.ICON_MONO_16x16 :
            if ((JLifeFormattedTextFieldBeanInfo.iconNameM16) == null) {
                return null;
            } else {
                if ((JLifeFormattedTextFieldBeanInfo.iconMono16) == null) {
                    JLifeFormattedTextFieldBeanInfo.iconMono16 = loadImage(JLifeFormattedTextFieldBeanInfo.iconNameM16);
                }
                return JLifeFormattedTextFieldBeanInfo.iconMono16;
            }
        case BeanInfo.ICON_MONO_32x32 :
            if ((JLifeFormattedTextFieldBeanInfo.iconNameM32) == null) {
                return null;
            } else {
                if ((JLifeFormattedTextFieldBeanInfo.iconMono32) == null) {
                    JLifeFormattedTextFieldBeanInfo.iconMono32 = loadImage(JLifeFormattedTextFieldBeanInfo.iconNameM32);
                }
                return JLifeFormattedTextFieldBeanInfo.iconMono32;
            }
        default :
            return null;
    }
}
