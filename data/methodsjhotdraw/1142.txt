/**
 * Convenience method for computing the total stroke width from the
 * STROKE_WIDTH, STROKE_INNER_WIDTH and STROKE_TYPE attributes.
 */
public static double getStrokeTotalWidth(Figure f) {
    switch (AttributeKeys.STROKE_TYPE.get(f)) {
        case BASIC :
        default :
            return AttributeKeys.STROKE_WIDTH.get(f);
            // break; not reached
        case DOUBLE :
            return (AttributeKeys.STROKE_WIDTH.get(f)) * (1.0 + (AttributeKeys.STROKE_INNER_WIDTH_FACTOR.get(f)));
            // break; not reached
    }
}
