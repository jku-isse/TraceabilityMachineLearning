public static Font getFont(Figure f) {
    Font prototype = AttributeKeys.FONT_FACE.get(f);
    if (prototype == null) {
        return null;
    }
    if ((AttributeKeys.getFontStyle(f)) != (Font.PLAIN)) {
        return prototype.deriveFont(AttributeKeys.getFontStyle(f), AttributeKeys.FONT_SIZE.get(f).floatValue());
    } else {
        return prototype.deriveFont(AttributeKeys.FONT_SIZE.get(f).floatValue());
    }
}
