public boolean contains(Point2D.Double p) {
    if ((AttributeKeys.TRANSFORM.get(this)) != null) {
        try {
            p = ((Point2D.Double) (AttributeKeys.TRANSFORM.get(this).inverseTransform(p, new Point2D.Double())));
        } catch (NoninvertibleTransformException ex) {
            InternalError error = new InternalError(ex.getMessage());
            error.initCause(ex);
            throw error;
        }
    }
    if (getDrawingArea().contains(p)) {
        for (Figure child : getChildrenFrontToBack()) {
            if ((child.isVisible()) && (child.contains(p))) {
                return true;
            }
        }
    }
    return false;
}
