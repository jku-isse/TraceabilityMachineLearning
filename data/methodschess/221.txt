/**
 * Add a transformation ply.
 *
 * @param source
 * 		The source square of the old piece.
 * @param destination
 * 		The destination square of the piece.
 * @param pieceType
 * 		The new piece type after the ply.
 * @param capture
 * 		Flag to indicate if another piece is captured.
 * @param score
 * 		The presort score of this ply.
 */
private final void addTransformationPly(int source, int destination, byte pieceType, boolean capture, short score) {
    addPly(new TransformationPlyImpl(new PositionImpl(source), new PositionImpl(destination), pieceType, capture), score);
}
