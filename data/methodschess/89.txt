public void actionPerformed(ActionEvent e) {
    for (byte col = 0; col <= 1; col++) {
        for (byte type = 1; type <= 6; type++) {
            if ((_pieceButton[col][type]) == (e.getSource())) {
                this.setType(type);
                this.setColor(col);
                this.jlSelectedPieceImage.setIcon(_pieceIcon[col][type]);
                return;
            }
        }
    }
}
