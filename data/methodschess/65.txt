/**
 * Return a new board, that results from a given ply.
 *
 * @param ply
 * 		The ply to perform.
 * @return A new board with the game position after the ply.
 */
public final Board getBoardAfterPly(Ply ply) {
    BitBoard newBoard = ((BitBoard) (clone()));
    newBoard.doPly(ply);
    return newBoard;
}
