// Constructors
/**
 * Create a new hashtable entry.
 *
 * @param board
 * 		The board before the ply.
 * @param ply
 * 		The ply to store.
 * @param searchDepth
 * 		The search depth, that was used to compute the ply.
 */
public PlyHashtableEntryImpl(Board board, Ply ply, int searchDepth) {
    setBoard(board);
    setPly(ply);
    setSearchDepth(searchDepth);
}
