/**
 * Sets the Dialog to the center of the screen
 *
 * @param dialog
 * 		The dialog which is to be centered on the screen
 */
public static void setDialogToCenter(JDialog dialog) {
    Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
    Dimension dialogSize = dialog.getSize();
    dialog.setLocation((((screenSize.width) - (dialogSize.width)) / 2), (((screenSize.height) - (dialogSize.height)) / 2));
}
