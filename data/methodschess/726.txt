// Constructors
/**
 * Create a new iterator instance.
 *
 * @param items
 * 		The array with the items.
 * @param nItems
 * 		The number of items (since the array is most likely longer).
 */
public ArrayStackIterator(Object[] items, int nItems) {
    _items = items;
    _nItems = nItems;
    _curElement = 0;
}
