/**
 * Stop the timer.
 */
public void stop() {
    // If there is a running thread.
    if ((_timerThread) != null) {
        Thread timerThread = _timerThread;// Copy it.

        _timerThread = null;
        // Signal to stop the thread.
        try {
            timerThread.join();// And wait for the thread to end.

        } catch (InterruptedException ignored) {
        }
    }
}
