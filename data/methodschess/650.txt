/**
 * Start the timer.
 */
public void start() {
    if ((_time) > 0L) {
        // if( _startTime == -1L) {
        // _startTime = System.currentTimeMillis();
        // }
        if ((_timerThread) == null) {
            _timerThread = new Thread(this);
            _timerThread.start();
        }
    }
}
