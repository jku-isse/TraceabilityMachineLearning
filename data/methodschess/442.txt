/**
 * Get the notation for a piece type.
 *
 * @return The notation for a piece type.
 */
public final String getPieceTypeNotation(byte pieceType) {
    String notation = "";// The result;

    // Convert the piece type to a string.
    switch (pieceType) {
        case Piece.PAWN :
            notation = "P";
            break;
        case Piece.KNIGHT :
            notation = "N";
            break;
        case Piece.BISHOP :
            notation = "B";
            break;
        case Piece.ROOK :
            notation = "R";
            break;
        case Piece.QUEEN :
            notation = "Q";
            break;
        case Piece.KING :
            notation = "K";
            break;
    }
    return notation;
}
