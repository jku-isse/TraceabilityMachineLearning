/**
 * Get the piece type from a figurine letter.
 *
 * @param letter
 * 		The figurine letter code.
 * @return The piece type or -1, if it was no valid figurine letter code.
 */
private final byte getPieceTypeFromLetter(char letter) {
    if (letter == P) {
        return Piece.PAWN;
    }
    if (letter == N) {
        return Piece.KNIGHT;
    }
    if (letter == B) {
        return Piece.BISHOP;
    }
    if (letter == R) {
        return Piece.ROOK;
    }
    if (letter == Q) {
        return Piece.QUEEN;
    }
    if (letter == K) {
        return Piece.KING;
    }
    return ((byte) (-1));
}
