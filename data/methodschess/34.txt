/**
 * Reset the game.
 */
public void reset() {
    // Reset all the game components.
    _game.reset();
    _gameNotation.reset();
    _board.initialPosition();
    _engine.reset();
    // Reset the timer to 40 min for each player.
    _gameTimer.reset((40 * 60));
    _controller.reset();
    _renderer.reset();
}
