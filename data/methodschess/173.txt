/**
 * Returns the status string to display in status bar
 */
public String getStatusDisplayString() {
    String retString = (((("HashSize: " + (this.getHashSize())) + " entries; ") + "Search time: ") + ((this.getMaximumSearchTime()) / 1000)) + "sec";
    if (this.isFixedTime()) {
        retString += " fix";
    } else {
        retString += " average";
    }
    return retString;
}
