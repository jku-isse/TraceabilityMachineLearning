// Methods
/**
 * Overridden  paintComponent method to draw the marker for
 * the source square.
 *
 * @param g
 * 		The graphics context.
 */
public void paintComponent(Graphics g) {
    if ((_sourceSquare) != (-1)) {
        // If a source square is selected,
        markSquare(_sourceSquare, g);// mark it.

    }
}
