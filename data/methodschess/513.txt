public final void tagPair() throws RecognitionException, TokenStreamException {
    match(PGNTokenTypes.LBRACK);
    whiteSpaces();
    {
        switch (LA(1)) {
            case PGNTokenTypes.TAG_EVENT :
                {
                    eventTag();
                    break;
                }
            case PGNTokenTypes.TAG_SITE :
                {
                    siteTag();
                    break;
                }
            case PGNTokenTypes.TAG_DATE :
                {
                    dateTag();
                    break;
                }
            case PGNTokenTypes.TAG_ROUND :
                {
                    roundTag();
                    break;
                }
            case PGNTokenTypes.TAG_WHITE :
                {
                    whiteTag();
                    break;
                }
            case PGNTokenTypes.TAG_BLACK :
                {
                    blackTag();
                    break;
                }
            case PGNTokenTypes.TAG_RESULT :
                {
                    resultTag();
                    break;
                }
            case PGNTokenTypes.TAG_OPENING :
                {
                    openingTag();
                    break;
                }
            default :
                {
                    throw new antlr.NoViableAltException(LT(1), getFilename());
                }
        }
    }
    whiteSpaces();
    match(PGNTokenTypes.RBRACK);
}
