/**
 * Returns a list of TransactionBeans between the two dates passed as params
 *
 * @param n
 * 		Number of "pages" of 20 log entries to retrieve.
 * @return list of 20*n TransactionBeans
 * @throws DBException
 * 		
 * @throws FormValidationException
 * 		
 */
public List<TransactionBean> getTransactions(Date time, int n) throws DBException, FormValidationException {
    List<PersonnelBean> dlhcps = patientDAO.getDeclaredHCPs(loggedInMID);
    // user has either 0 or 1 DLHCPs. Get one if exists so it can be filtered from results
    long dlhcpID = -1;
    if (!(dlhcps.isEmpty()))
        dlhcpID = dlhcps.get(0).getMID();

    List<TransactionBean> fullList = transDAO.getTransactionsAffecting(loggedInMID, dlhcpID, time, ((20 * n) + 1));
    return fullList;
}
