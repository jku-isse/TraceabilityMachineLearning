/**
 * Takes the information out of the PatientBean param and updates the patients information
 *
 * @param p
 * 		the new patient information
 * @throws iTrustException
 * 		
 * @throws FormValidationException
 * 		
 */
public void updateInformation(PatientBean p) throws FormValidationException, iTrustException {
    p.setMID(pid);// for security reasons

    validator.validate(p);
    patientDAO.editPatient(p, loggedInMID);
    emailutil.sendEmail(makeEmail());
}
