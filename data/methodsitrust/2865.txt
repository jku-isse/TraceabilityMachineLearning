/**
 * Change the password of a particular user
 *
 * @param mid
 * 		The MID of the user whose password we are changing.
 * @param password
 * 		The new password.
 * @throws DBException
 * 		
 */
public void resetPassword(long mid, String password) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement("UPDATE users SET password=? WHERE MID=?");
        password = DigestUtils.shaHex(password);
        ps.setString(1, password);
        ps.setLong(2, mid);
        ps.executeUpdate();
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
