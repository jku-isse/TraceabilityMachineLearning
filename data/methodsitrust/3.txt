/**
 * Checks for overloaded methods
 *
 * @param bean
 * 		item to check
 */
private void checkOverloadedMethods(T bean) {
    Method[] methods = bean.getClass().getDeclaredMethods();
    HashMap<String, String> nameMap = new HashMap<String, String>(methods.length);
    for (Method method : methods) {
        if ((nameMap.get(method.getName())) != null)
            throw new IllegalArgumentException((((bean.getClass().getName()) + " should not have any overloaded methods, like ") + (method.getName())));

        // allow an equals, compareTo override
        if (!(("equals".equals(method.getName())) || ("compareTo".equals(method.getName()))))
            nameMap.put(method.getName(), "exists");

    }
}
