/**
 * Return a list of patients who share at least one parent (and that parent is not 0) with this patient
 *
 * @param pid
 * 		-
 * 		this patient
 * @return A java.util.list of FamilyMemberBeans.
 * @throws DBException
 * 		
 */
public List<FamilyMemberBean> getSiblings(long pid) throws DBException {
    return getFamilyMembers(pid, "Sibling", ("SELECT p2.FirstName AS FirstName, p2.LastName AS LastName, p2.MID as MID " + ((("FROM patients p1, patients p2 " + "WHERE p1.MID=? AND p1.MID<>p2.MID ") + "AND( (p1.MotherMID=p2.MotherMID AND p2.MotherMID<>0)") + "  OR (p1.FatherMID=p2.FatherMID AND p1.FatherMID<>0))")), false);
}
