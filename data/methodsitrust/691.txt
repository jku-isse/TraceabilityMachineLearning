/**
 * Set up defaults
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param ipAddr
 * 		The IP address of the user making the login attempt.
 */
public LoginFailureAction(DAOFactory factory, String ipAddr) {
    this.authDAO = factory.getAuthDAO();
    this.ipAddr = ipAddr;
    validCaptcha = false;
    hasAttempts = false;
}
