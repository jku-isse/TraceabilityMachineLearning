public List<RiskChecker> getDiseaseAtRisk() {
    List<RiskChecker> diseases = new ArrayList<RiskChecker>();
    for (RiskChecker diseaseChecker : riskCheckers) {
        if (diseaseChecker.isAtRisk())
            diseases.add(diseaseChecker);

    }
    return diseases;
}
