/**
 * "Send" an email, which just inserts it into the database.
 *
 * @param email
 * 		The Email object to insert.
 * @throws DBException
 * 		
 */
public void sendEmailRecord(Email email) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement(("INSERT INTO fakeemail (ToAddr, FromAddr, Subject, Body) " + "VALUES (?,?,?,?)"));
        emailBeanLoader.loadParameters(ps, email);
        ps.executeUpdate();
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
