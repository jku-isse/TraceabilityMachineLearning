/**
 * Set up defaults.
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the person retrieving survey results.
 */
public ViewSurveyResultAction(DAOFactory factory, long loggedInMID) {
    surveyResultDAO = factory.getSurveyResultDAO();
    this.loggedInMID = loggedInMID;
    validator = new SurveySearchValidator();
}
