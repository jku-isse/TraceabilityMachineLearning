 * @param patient
 * 		
 * @param filterType
 * 		
 * @return 
 */
public String getComprehensivePersonnelInfo(PatientBean patient, PersonnelReportFilter.PersonnelReportFilterType filterType) {
    switch (filterType) {
        case DLHCP :
            try {
                List<PersonnelBean> dlhcps = pDAO.getDeclaredHCPs(patient.getMID());
                String out = "";
                for (PersonnelBean dlhcp : dlhcps) {
                    out += (dlhcp.getFullName()) + "
";
                }
                return out;
            } catch (Exception e) {
                break;
            }
        default :
            break;
    }
    return null;
}
