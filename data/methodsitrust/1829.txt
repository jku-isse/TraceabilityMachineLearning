public int getAge() {
    try {
        long ageInMs = (System.currentTimeMillis()) - (new SimpleDateFormat("MM/dd/yyyy").parse(dateOfBirthStr).getTime());
        long age = ageInMs / ((((1000L * 60L) * 60L) * 24L) * 365L);
        return ((int) (age));
    } catch (ParseException e) {
        e.printStackTrace();
        return -1;
    }
}
