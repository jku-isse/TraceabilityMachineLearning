/**
 *
 *
 * @param name
 * 		
 * @return 
 */
public static DemographicReportFilter.DemographicReportFilterType filterTypeFromString(String name) {
    for (DemographicReportFilter.DemographicReportFilterType type : DemographicReportFilter.DemographicReportFilterType.values()) {
        if (type.name().equalsIgnoreCase(name)) {
            return type;
        }
    }
    return null;
}
