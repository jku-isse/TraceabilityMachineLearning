public void testRepresent() throws Exception {
    // Login
    WebConversation wc = login("2", "pw");
    WebResponse wr = wc.getCurrentPage();
    iTrustHTTPTest.assertLogged(TransactionType.HOME_VIEW, 2L, 0L, "");
    wr = wr.getLinkWith("View My Records").click();
    iTrustHTTPTest.assertLogged(TransactionType.MEDICAL_RECORD_VIEW, 2L, 2L, "");
    wr = wr.getLinkWith("Baby Programmer").click();
    // Clicking on a representees name takes you to their records
    assertTrue(wr.getText().contains("You are currently viewing your representees records"));
    iTrustHTTPTest.assertLogged(TransactionType.MEDICAL_RECORD_VIEW, 2L, 5L, "");
}
