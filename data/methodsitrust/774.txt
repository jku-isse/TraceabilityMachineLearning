/**
 * Super class validates pidString
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the user who is making a prescription report.
 * @param pidString
 * 		The MID of the patient in question.
 * @throws iTrustException
 * 		
 * @throws DBException
 * 		
 * @throws NoHealthRecordsException
 * 		
 */
public PrescriptionReportAction(DAOFactory factory, long loggedInMID, String pidString) throws DBException, NoHealthRecordsException, iTrustException {
    super(factory, pidString);
    this.ovDAO = factory.getOfficeVisitDAO();
    this.patientDAO = factory.getPatientDAO();
    this.prDAO = factory.getPrescriptionReportDAO();
    this.loggedInMID = loggedInMID;
}
