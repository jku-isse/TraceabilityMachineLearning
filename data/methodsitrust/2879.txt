public Date findEarliestIncident(String icdCode) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement(("SELECT MIN(visitDate) AS visitDate FROM ovdiagnosis d INNER JOIN officevisits o " + (" ON d.VisitID=o.ID " + " WHERE ICDCode LIKE ?")));
        ps.setString(1, (icdCode + "%"));
        ResultSet rs = ps.executeQuery();
        if (rs.next()) {
            Date date = rs.getDate("visitDate");
            return date;
        }
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
    return null;
}
