/**
 * Close the prepared statement and the connection in a proper way
 *
 * @param conn
 * 		
 * @param ps
 * 		
 */
public static void closeConnection(Connection conn, PreparedStatement ps) {
    try {
        if (ps != null)
            ps.close();

        if (conn != null)
            conn.close();

    } catch (SQLException e) {
        System.err.println("Error closing connections");
        e.printStackTrace();
    }
}
