public void setSmoker(int smoker) {
    this.smokingStatus = smoker;
    switch (smoker) {
        case 1 :
            this.isSmoker = true;
            this.smokingStatusDesc = "Current every day smoker";
            break;
        case 2 :
            this.isSmoker = true;
            this.smokingStatusDesc = "Current some day smoker";
            break;
        case 3 :
            this.isSmoker = false;
            this.smokingStatusDesc = "Former smoker";
            break;
        case 4 :
            this.isSmoker = false;
            this.smokingStatusDesc = "Never smoker";
            break;
        case 5 :
            this.isSmoker = true;
            this.smokingStatusDesc = "Smoker, current status unknown";
            break;
        case 9 :
            this.isSmoker = false;
            this.smokingStatusDesc = "Unknown if ever smoked";
            break;
    }
}
