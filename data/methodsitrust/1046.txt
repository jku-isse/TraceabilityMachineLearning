    assertEquals(5, accesses.size());
    for (TransactionBean t : accesses) {
        assertEquals(9000000000L, t.getLoggedInMID());
        assertEquals(2L, t.getSecondaryMID());
        assertEquals("Viewed patient records", t.getAddedInfo());
    }
