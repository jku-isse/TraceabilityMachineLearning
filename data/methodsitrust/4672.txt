/**
 * Check against the proper gender
 *
 * @param name
 * 		
 * @param gen
 * 		
 * @param format
 * 		
 * @param isNullable
 * 		
 * @return 
 */
protected String checkGender(String name, Gender gen, ValidationFormat format, boolean isNullable) {
    String str = "";
    if (gen != null)
        str = gen.toString();

    return checkFormat(name, str, format, isNullable);
}
