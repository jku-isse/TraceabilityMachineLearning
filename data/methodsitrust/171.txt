 * Adds a patients telemedicine data to the database.
 *

 * 		
 */
public void addRemoteMonitoringData(RemoteMonitoringDataBean rmdBean) throws DBException, FormValidationException, iTrustException {
    validator.validate(rmdBean);
    String role;
    if ((loggedInMID) == (patientMID)) {
        role = "self-reported";
    } else
        if (authDAO.getUserRole(loggedInMID).getUserRolesString().equals("uap")) {
            role = "case-manager";
        } else {
            role = "patient representative";
        }

    // Store in DB
    rmDAO.storePatientData(patientMID, rmdBean, role, loggedInMID);
}
