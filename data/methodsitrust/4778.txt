public void testValidateBadName() {
    bean.setName("");// empty

    try {
        validator.validate(bean);
        fail("Expected a validation failure.");
    } catch (FormValidationException e) {
        // passed.
    }
    bean.setName("@");// bad symbol

    try {
        validator.validate(bean);
        fail("Expected a validation failure.");
    } catch (FormValidationException e) {
        // passed.
    }
    bean.setName(">");// bad symbol

    try {
        validator.validate(bean);
        fail("Expected a validation failure.");
    } catch (FormValidationException e) {
        // passed.
    }
    bean.setName(longString(101));// string too long

    try {
        validator.validate(bean);
        fail("Expected a validation failure.");
    } catch (FormValidationException e) {
        // passed.
    }
}
