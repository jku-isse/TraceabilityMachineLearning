/**
 * Constructor
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the HCP retrieving the patient data.
 */
public ViewMyRemoteMonitoringListAction(DAOFactory factory, long loggedInMID) {

    this.rmDAO = factory.getRemoteMonitoringDAO();
