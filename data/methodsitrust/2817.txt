/**
 * Adds a message to the database.
 *
 * @param mBean
 * 		A bean representing the message to be added.
 * @throws SQLException
 * 		
 * @throws DBException
 * 		
 */
public void addReport(AdverseEventBean aeBean, long hcpmid) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement(("INSERT INTO adverseevents (PatientMID, PresImmu, Code, Comment, Prescriber, Status) " + "VALUES (?, ?, ?, ?, ?, ?)"));
        ps.setString(1, aeBean.getMID());
        ps.setString(2, aeBean.getDrug());
        ps.setString(3, aeBean.getCode());
        ps.setString(4, aeBean.getDescription());
        ps.setLong(5, hcpmid);
        ps.setString(6, "Active");
        ps.executeUpdate();
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
