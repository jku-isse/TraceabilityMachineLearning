/**
 * Adds a prescription bean to the database.
 *
 * @param pres
 * 		The prescription bean to be added.
 * @return The unique ID of the newly added bean.
 * @throws DBException
 * 		
 */
public long add(PrescriptionBean pres) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement("INSERT INTO ovmedication (VisitID,NDCode,StartDate,EndDate,Dosage,Instructions,OverrideOther) VALUES (?,?,?,?,?,?,?)");
        loader.loadParameters(ps, pres);
        ps.executeUpdate();
        return DBUtil.getLastInsert(conn);
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
