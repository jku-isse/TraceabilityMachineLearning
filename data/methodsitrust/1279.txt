/**
 * Set up defaults
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the person viewing the report.
 */
public ViewReportAction(DAOFactory factory, long loggedInMID) {
    patientDAO = factory.getPatientDAO();
    personnelDAO = factory.getPersonnelDAO();
}
