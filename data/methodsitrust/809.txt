/**
 * Checks if the given mid matches the given role
 *
 * @param mid
 * 		the mid to be checked
 * @param role
 * 		the role to be checked
 * @return true if the mid and role match
 * @throws iTrustException
 * 		
 */
public String checkRole(long mid, String role) throws iTrustException {
    try {
        if ((((((("patient".equals(role)) && (patientDAO.getRole(mid, role).equals("patient"))) || (("hcp".equals(role)) && (patientDAO.getRole(mid, role).equals("hcp")))) || (("uap".equals(role)) && (patientDAO.getRole(mid, role).equals("uap")))) || (("pha".equals(role)) && (patientDAO.getRole(mid, role).equals("pha")))) || (("er".equals(role)) && (patientDAO.getRole(mid, role).equals("er")))) || (("lt".equals(role)) && (patientDAO.getRole(mid, role).equals("lt"))))
            return role;
        else
            return null;

    } catch (DBException e) {
        e.printStackTrace();
    } catch (iTrustException e) {
        e.printStackTrace();
        throw e;
    }
    return null;
}
