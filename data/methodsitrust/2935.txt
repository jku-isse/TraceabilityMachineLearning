/**
 * Get a count of all pending lab procedures for a particular HCP.
 *
 * @param mid
 * 		
 * @return 
 * @throws DBException
 * 		
 */
public int getHCPPendingCount(long mid) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        int count = 0;
        conn = factory.getConnection();
        ps = conn.prepareStatement(("SELECT COUNT(*) FROM labprocedure WHERE Status = ? AND labprocedure.OfficeVisitID IN " + "(SELECT officevisits.ID FROM officevisits WHERE officevisits.HCPID = ?)"));
        ps.setString(1, LabProcedureBean.Pending);
        ps.setLong(2, mid);
        ResultSet rs = ps.executeQuery();
        if (rs.next()) {
            count = rs.getInt(1);
        }
        return count;
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
