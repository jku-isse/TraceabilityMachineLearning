/**
 * Returns all of the hospitals this LHCP is associated with.
 *
 * @param mid
 * 		The MID of the personnel in question.
 * @return A java.util.List of HospitalBeans.
 * @throws DBException
 * 		
 */
public List<HospitalBean> getHospitals(long mid) throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement(("SELECT * FROM hcpassignedhos hah,hospitals h " + "WHERE hah.HCPID=? AND hah.HosID=h.HospitalID ORDER BY HospitalName ASC"));
        ps.setLong(1, mid);
        ResultSet rs = ps.executeQuery();
        return hospitalBeanLoader.loadList(rs);
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
