public void executeSQL(List<String> queries) throws SQLException {
    Connection conn = factory.getConnection();
    long start = System.currentTimeMillis();
    for (String sql : queries) {
        System.out.println(("Executing: " + sql));
        // System.out.println(""" + sql.replace(""", "\") + "",");
        (DBBuilder.numExecuted)++;
        Statement stmt = conn.createStatement();
        try {
            stmt.execute(sql);
        } catch (SQLException e) {
            throw new SQLException((((e.getMessage()) + " from executing: ") + sql), e.getSQLState(), e.getErrorCode());
        } finally {
            stmt.close();
        }
    }
    DBBuilder.queryTimeTaken += (System.currentTimeMillis()) - start;
    System.out.println((((("Queries executed: " + (DBBuilder.numExecuted)) + "
Time executing queries: ") + (DBBuilder.queryTimeTaken)) + "ms"));
    conn.close();
}
