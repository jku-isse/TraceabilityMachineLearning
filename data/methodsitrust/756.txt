/**
 * Compares one HCP with another
 *
 * @param a
 * 		the first HCP
 * @param b
 * 		the second HCP
 * @return -1 if a has had more patients, 1 if b has had more patients; otherwise 0
 */
public int compare(HCPDiagnosisBean a, HCPDiagnosisBean b) {
    int ret = 0;
    if ((a.getNumPatients()) > (b.getNumPatients()))
        ret = -1;
    else
        if ((a.getNumPatients()) < (b.getNumPatients()))
            ret = 1;


    return ret;
}
