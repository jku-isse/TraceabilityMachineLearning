/**
 * Set up defaults
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the person viewing the expired prescriptions.
 */
public ViewExpiredPrescriptionsAction(DAOFactory factory, long loggedInMID) {
    this.emailer = new EmailUtil(factory);
    this.patientDAO = factory.getPatientDAO();
    this.personnelDAO = factory.getPersonnelDAO();
    this.loggedInMID = loggedInMID;
}
