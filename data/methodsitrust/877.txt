/**
 * Sets up defaults
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the user taking the survey.
 */
public SurveyAction(DAOFactory factory, long loggedInMID) {
    surveyDAO = factory.getSurveyDAO();
    ovDAO = factory.getOfficeVisitDAO();
    this.loggedInMID = loggedInMID;
}
