/**
 * Returns the operation profile
 *
 * @return The OperationalProfile as a bean.
 * @throws DBException
 * 		
 */
public OperationalProfile getOperationalProfile() throws DBException {
    Connection conn = null;
    PreparedStatement ps = null;
    try {
        conn = factory.getConnection();
        ps = conn.prepareStatement(("SELECT TransactionCode, count(transactionID) as TotalCount, " + (("count(if(loggedInMID<9000000000, transactionID, null)) as PatientCount, " + "count(if(loggedInMID>=9000000000, transactionID, null)) as PersonnelCount ") + "FROM transactionlog GROUP BY transactionCode ORDER BY transactionCode ASC")));
        ResultSet rs = ps.executeQuery();
        return operationalProfileLoader.loadSingle(rs);
    } catch (SQLException e) {
        e.printStackTrace();
        throw new DBException(e);
    } finally {
        DBUtil.closeConnection(conn, ps);
    }
}
