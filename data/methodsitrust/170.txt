/**
 * Constructor
 *
 * @param factory
 * 		The DAOFactory used to create the DAOs used in this action.
 * @param loggedInMID
 * 		The MID of the person recording the patients data.
 * @param patientMID
 * 		The MID of the patient
 */
public AddRemoteMonitoringDataAction(DAOFactory factory, long loggedInMID, long patientMID) {
    this.loggedInMID = loggedInMID;
    this.rmDAO = factory.getRemoteMonitoringDAO();
    this.authDAO = factory.getAuthDAO();
    this.patientMID = patientMID;
}
