public void addApptRequest(ApptRequestBean req) throws SQLException {
    Connection conn = null;
    PreparedStatement stmt = null;
    conn = factory.getConnection();
    stmt = conn.prepareStatement("INSERT INTO appointmentrequests (appt_type, patient_id, doctor_id, sched_date, comment, pending, accepted) VALUES (?, ?, ?, ?, ?, ?, ?)");
    loader.loadParameters(stmt, req);
    stmt.executeUpdate();

}
